    ```python
    import asyncio
    import logging
    from datetime import datetime, timezone

    from quart import Quart, jsonify, request
    from quart_cors import cors
    from hypercorn.config import Config as HypercornConfig
    from hypercorn.asyncio import serve

    from .config import settings
    from .models import Node

    logger = logging.getLogger(__name__)

    app = Quart(__name__)
    app = cors(app, allow_origin="*")

    # In-memory state (per CONTEXT.md locked decision — dict keyed by IP string, asyncio.Lock)
    node_registry: dict[str, Node] = {}
    registry_lock = asyncio.Lock()
    ready_flag = False  # becomes True after @app.before_serving completes

    @app.route("/livez")
    async def livez():
        """Liveness endpoint — always responds if process is running (per CONTEXT.md).
        Returns 200 OK {"status": "alive"}"""
        return jsonify({"status": "alive"}), 200

    @app.route("/readyz")
    async def readyz():
        """Readiness endpoint — 200 when leader accepts registrations (per CONTEXT.md).
        Returns 200 OK when ready_flag is True, 503 otherwise."""
        if ready_flag:
            return jsonify({"status": "ready"}), 200
        else:
            return jsonify({"status": "not ready"}), 503

    @app.errorhandler(Exception)
    async def handle_exception(e):
        """Global error handler returning JSON format per CONTEXT.md: {"error": "...", "status": NNN}"""
        logger.exception("Unhandled exception")
        return jsonify({"error": str(e), "status": 500}), 500

    async def startup():
        """Runs on server start: set ready flag, log startup (per CONTEXT.md @app.before_serving pattern)."""
        global ready_flag
        logger.info(f"Leader starting on {settings.host}:{settings.port}")
        ready_flag = True

    app.before_serving(startup)

    def get_hypercorn_config() -> HypercornConfig:
        """Build Hypercorn config inline in leader.py __main__ block pattern (per CONTEXT.md locked decision)."""
        hconf = HypercornConfig()
        hconf.bind = [f"{settings.host}:{settings.port}"]
        hconf.loglevel = settings.log_level.lower()
        return hconf

    def main():
        logging.basicConfig(
            level=getattr(logging, settings.log_level),
            format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
            stream=__import__("sys").stderr,  # per CONTEXT.md: logging to stderr
        )
        hconf = get_hypercorn_config()
        asyncio.run(serve(app, hconf))  # type: ignore[arg-type]

    if __name__ == "__main__":
        main()
    ```
