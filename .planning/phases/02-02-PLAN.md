### 2. Modify `run_checks` to retry before new checks
At the START of the `while True` loop, BEFORE fetching peers, drain the buffer:

```python
# Retry buffered submissions first (per NODE-06)
if _buffer:
    logger.info("Retrying %d buffered submissions", len(_buffer))
    retry_results = []
    while _buffer:
        payload = _buffer.popleft()
        success = await submit_results(payload["node_ip"], leader_ip, port, payload["checks"])
        if not success:
            _buffer.appendleft(payload)  # put back if still failing
            break  # stop retrying on first failure — try again next cycle
        retry_results.append(payload)
    if retry_results:
        logger.info("Retried %d buffered submissions successfully", len(retry_results))
```

Key decisions:
- Buffer is a `deque` with no fixed maxlen (we manage overflow manually with FIFO eviction per D-10)
- On buffer overflow (len > _buffer_size): `_buffer.popleft()` evicts oldest (FIFO per D-10)
- Retry stops on first failure — subsequent items stay buffered for next cycle
- Buffered payloads that succeed are removed from the deque
- A failed retry keeps the payload in the deque for the next cycle

3. **Add `async def _handle_http_request(reader, writer)` in node.py:**
```python
async def _handle_http_request(reader: asyncio.StreamReader, writer: asyncio.StreamWriter):
    """Handle incoming HTTP requests on node's port."""
    try:
        request_line = await asyncio.wait_for(reader.readline(), timeout=5)
        if not request_line:
            writer.close()
            return

        request_text = request_line.decode("utf-8", errors="replace").strip()
        parts = request_text.split(" ")
        if len(parts) < 2:
            writer.close()
            return

        method = parts[0]
        path = parts[1]

        # Read headers
        content_length = 0
        while True:
            header_line = await asyncio.wait_for(reader.readline(), timeout=5)
            header_str = header_line.decode("utf-8", errors="replace").strip()
            if not header_str:
                break
            if header_str.lower().startswith("content-length:"):
                content_length = int(header_str.split(":")[1].strip())

        # Read body if present
        body = b""
        if content_length > 0:
            body = await asyncio.wait_for(reader.readexactly(content_length), timeout=5)

        if method == "POST" and path == "/updateConfig":
            data = json.loads(body.decode("utf-8"))
            await _handle_update_config(data)
            response = b"HTTP/1.1 200 OK\r\nContent-Type: application/json\r\nContent-Length: 15\r\n\r\n{\"status\":\"ok\"}"
        elif method == "POST" and path == "/update-peers":
            # Silently accept peer push from leader (Phase 1 leader pushes this)
            response = b"HTTP/1.1 200 OK\r\nContent-Type: application/json\r\nContent-Length: 15\r\n\r\n{\"status\":\"ok\"}"
        else:
            response = b"HTTP/1.1 404 Not Found\r\nContent-Length: 0\r\n\r\n"

        writer.write(response)
        await writer.drain()
    except (asyncio.TimeoutError, json.JSONDecodeError, Exception) as e:
        logger.debug("HTTP handler error: %s", e)
    finally:
        try:
            writer.close()
        except Exception:
            pass
```
