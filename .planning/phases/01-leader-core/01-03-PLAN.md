    ```python
    #!/usr/bin/env python3
    """Register a node with the mesh-status leader.
    
    Usage:
        python register.py --node-ip 10.0.0.1 --leader-ip 10.0.0.2
        python register.py  # prompts via stdin
        
    Per CONTEXT.md: standalone CLI script (not part of mesh_status package).
    Sends POST /register to leader and prints peer list response.
    """

    import argparse
    import json
    import sys
    from urllib.request import Request, urlopen
    from urllib.error import URLError

    LEADER_PORT = 58080


    def parse_args(argv: list[str] | None = None) -> argparse.Namespace:
        """Parse command line arguments for --node-ip and --leader-ip (per REGI-01)."""
        parser = argparse.ArgumentParser(
            description="Register a node with the mesh-status leader."
        )
        parser.add_argument(
            "--node-ip",
            help="IP address of this node (e.g., 10.0.0.1)",
        )
        parser.add_argument(
            "--leader-ip",
            help="IP address of the leader (e.g., 10.0.0.2)",
        )
        return parser.parse_args(argv)


    def prompt_stdin() -> tuple[str, str]:
        """Prompt for node-ip and leader-ip via stdin (per REGI-02).
        
        Falls back to interactive input() when argv values are not provided.
        """
        node_ip = input("Enter this node's IP address: ").strip()
        leader_ip = input("Enter the leader's IP address: ").strip()
        return node_ip, leader_ip


    def register(node_ip: str, leader_ip: str) -> dict:
        """Send POST /register to the leader and return the parsed JSON response.
        
        Per CONTEXT.md:
        - Payload: {"node_ip": "...", "hostname": "..."}
        - Endpoint: http://{leader_ip}:58080/register
        - Response: {"status": "registered", "peers": [...]}
        """
        url = f"http://{leader_ip}:{LEADER_PORT}/register"
        payload = json.dumps({"node_ip": node_ip, "hostname": ""}).encode("utf-8")
        req = Request(url, data=payload, headers={"Content-Type": "application/json"})
        try:
            with urlopen(req, timeout=10) as resp:
                return json.loads(resp.read().decode("utf-8"))
        except URLError as e:
            print(f"Error: Could not reach leader at {url}: {e}", file=sys.stderr)
            sys.exit(1)
        except json.JSONDecodeError as e:
            print(f"Error: Invalid JSON response from leader: {e}", file=sys.stderr)
            sys.exit(1)


    def main() -> None:
        """Entry point — parse args, prompt if needed, register, display result."""
        args = parse_args()
        node_ip = args.node_ip
        leader_ip = args.leader_ip

        # Stdin fallback per REGI-02: when either arg is omitted
        if not node_ip or not leader_ip:
            if not node_ip:
                print("No --node-ip provided, prompting for input...")
            if not leader_ip:
                print("No --leader-ip provided, prompting for input...")
            stdin_node_ip, stdin_leader_ip = prompt_stdin()
            node_ip = node_ip or stdin_node_ip
            leader_ip = leader_ip or stdin_leader_ip

        if not node_ip or not leader_ip:
            print("Error: Both node-ip and leader-ip are required.", file=sys.stderr)
            sys.exit(1)

        print(f"Registering node {node_ip} with leader at {leader_ip}:{LEADER_PORT}...")
        result = register(node_ip, leader_ip)
        print(f"Status: {result.get('status', 'unknown')}")
        peers = result.get("peers", [])
        if peers:
            print(f"Peers ({len(peers)}):")
            for peer in peers:
                print(f"  - {peer}")
        else:
            print("Peers: (none — first node in mesh)")
        print("Registration complete.")
        sys.exit(0)


    if __name__ == "__main__":
        main()
    ```

    - `--node-ip` and `--leader-ip` via argv (REGI-01) ✓
    - stdin fallback via `input()` when args omitted (REGI-02) ✓
    - Payload: `{"node_ip": node_ip, "hostname": ""}` (optional hostname field per CONTEXT.md) ✓
    - Endpoint: `http://{leader_ip}:58080/register` (port 58080 per LEAD-01) ✓
    - Standalone script (not in mesh_status package) per CONTEXT.md specifics section ✓
    - Uses `urllib.request` from stdlib (no external dependencies needed — script must work without requiring pip install) ✓

    **Note on dependency choice:** Using stdlib `urllib.request` instead of `httpx`/`requests`. The `register.py` is standalone per CONTEXT.md — it should work without the package being installed. This avoids requiring the user to `pip install -e .` before they can register. The leader package itself uses `httpx` for async peer push.
